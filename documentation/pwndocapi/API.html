 32    def __init__(self, host, port=443, ssl=True, verbose=True):
 33        super(API, self).__init__()
 34        if ssl == True:
 35            self.target = "https://%s:%d" % (host, port)
 36        else:
 37            self.target = "http://%s:%d" % (host, port)
 38        self.session = requests.Session()
 39        self.loggedin = False
 40        self.verbose = verbose
 41        self.user = {"token": "", "refreshToken": ""}
 42
 43    def login(self, username, password):
 44        r = self.session.post(
 45            self.target + "/api/users/token",
 46            json={"username": username, "password": password, "totpToken": ""},
 47            verify=False
 48        )
 49        if r.json()["status"] == "success":
 50            if self.verbose:
 51                print("[>] Successfully logged in.")
 52            self.user = {
 53                "token": r.json()["datas"]["token"],
 54                "refreshToken": r.json()["datas"]["refreshToken"]
 55            }
 56            self.session.cookies.set("token", "JWT%20" + self.user["token"])
 57            self.loggedin = True
 58        elif r.json()["status"] == "error":
 59            if self.verbose:
 60                print("[!] Login error. (%s)" % r.json()["status"])
 61            self.loggedin = False
 62        return self.loggedin
